void matvec(double** A, double* p, double* Ap) {
  int row, col;
  for (row = 0; row < N; ++row) {
    Ap[row] = 0;
    for (col = 0; col < N; ++col) {
      Ap[row] += A[row][col] * p[col]; corruptDouble(VAR_INFO_ONLY(Ap),&Ap[row]);
    }
  }
}

void waxpby(double a, double* x, double b, double* y, double* r) {
  int i;
  for (i = 0; i < N; ++i) {
    r[i] = a * x[i] + b * y[i]; corruptDouble(VAR_INFO_ONLY(r),&r[i]);
  }
}

void daxpby(double a, double* x, double b, double* y) {
  int i;
  for (i = 0; i < N; ++i) {
    y[i] = a * x[i] + b * y[i]; corruptDouble(VAR_INFO_ONLY(y),&y[i]);
  }
}

double dot_r2(double* r) {
  double sum = 0;
  int i;
  for (i = 0; i < N; ++i) {
    sum += r[i]*r[i]; corruptDouble(VAR_INFO_ONLY(sum),&sum);
  }
  return sum;
}

double dot(double* x, double* y) {
  double sum = 0;
  int i;
  for (i = 0; i < N; ++i) {
    sum += x[i]*y[i]; corruptDouble(VAR_INFO_ONLY(sum),&sum);
  }
  return sum;
}

void solve_cg(double** A, double* b, double* x, double* Ap, double* p, double* r) {
  double normr;
  double rtrans = 0, oldrtrans = 0;
  double p_ap_dot = 0;
  //double alpha = 0;
  double alpha = 0;
  double tolerance = 0.06;
  //double tolerance = 2.22045e-16;
  int k;

  waxpby(1, x, 0, x, p);
  matvec(A, p, Ap);
  waxpby(1, b, -1, Ap, r);
  rtrans = dot_r2(r); corruptDouble(VAR_INFO_ONLY(rtrans),&rtrans);
  normr = sqrt(rtrans); corruptDouble(VAR_INFO_ONLY(normr), &normr);

  for (k = 1; k <= MAX_ITER && normr > tolerance; ++k) {
    if (k == 1) {
      waxpby(1, r, 0, r, p);
    } else {
      oldrtrans = rtrans; corruptDouble(VAR_INFO_ONLY(oldrtrans), &oldrtrans);
      rtrans = dot_r2(r); corruptDouble(VAR_INFO_ONLY(rtrans), &rtrans);
      double beta = rtrans/oldrtrans; corruptDouble(VAR_INFO_ONLY(beta), &beta);
      daxpby(1, r, beta, p);
    }
    normr = sqrt(rtrans); corruptDouble(VAR_INFO_ONLY(normr), &normr);

    matvec(A, p, Ap);
    p_ap_dot = dot(Ap, p); corruptDouble(VAR_INFO_ONLY(p_ap_dot), &p_ap_dot);
    alpha = rtrans/p_ap_dot; corruptDouble(VAR_INFO_ONLY(alpha), &alpha);
    daxpby(alpha, p, 1, x);
    daxpby(-alpha, Ap, 1, r);
  }

  double golden=1.4175e-02;
  printf("normr: %.4e\n",normr);
  printf("iterations: %d\n",k);
  logTestResult("normrdiff", fabs(golden-normr));
  //printVec(x);
}

void readA(double** A, FILE *fp) {
  int i, j;

  for (i = 0; i < N; i++) {
    for (j = 0; j < N; j++) {
      fscanf(fp, "%lf", &A[i][j]); corruptDouble(VAR_INFO_ONLY(A),&A[i][j]);
    }
  }
}

void readB(double* b, FILE *fp) {
  int i;
  for (i = 0; i < N; i++) {
    fscanf(fp, "%lf", &b[i]); corruptDouble(VAR_INFO_ONLY(b),&b[i]);
  }
}
